POST /v1/visual/search

Text-to-image visual search.

{
  "query": "apple",
  "limit": 12
}

{
  "items": [
    { "recordId": "/2021009/_ULAN_500115493", "score": 0.83 },

  "meta": { "query": "apple", "limit": 12 }
}
